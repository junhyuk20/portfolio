create table m_table(
	member_no	number			not null
	,member_id	varchar2(100) 	primary key 
	,member_pw	varchar2(50)	not null
	,member_nm	varchar2(50)	not null
)
create sequence m_table_seq;



create table b_table(
	board_no		number			primary key
	,member_id		varchar2(100)	not null 
	,board_title	varchar2(1000)	not null
	,board_context	varchar2(2000)	not null
	,board_indate	date			default	sysdate	
	,board_hits		number			default	0
	,originalfile	varchar2(1000)
	,savedfile		varchar2(1000)
)
create sequence b_table_seq;
alter table b_table add constraint b_table_fk foreign key(member_id) references m_table(member_id);


create table c_table(
	comment_no	number	primary key
	,board_no	number	not null constraint c_table_fk1 references b_table(board_no)
	,member_nm 	varchar2(50) not null
	,member_id	varchar2(100)	not null constraint c_table_fk2 references m_table(member_id)
	,comment_contents	varchar2(1000)	not null
	,comment_indate 	date	default sysdate
	,comment_parent	number	default 0 
);
create sequence c_table_seq;
alter table c_table add constraint c_table_fk1 foreign key(member_id) references m_table(member_id);
alter table c_table add constraint c_table_fk2 foreign key(board_no) references b_table(board_no);
